                statusMessage.innerHTML = `<div class="status-message status-success">${translations.successMessage[currentLanguage]}</div>`;
                
                // Reset form
                document.getElementById('contactForm').reset();
                
            } catch (error) {
                // Show error message
                statusMessage.innerHTML = `<div class="status-message status-error">${translations.errorMessage[currentLanguage]}</div>`;
            } finally {
                // Reset button
                submitBtn.disabled = false;
                submitBtn.innerHTML = '<i class="fas fa-paper-plane"></i><span data-translate="send">' + translations.send[currentLanguage] + '</span>';
                
                // Clear status message after 5 seconds
                setTimeout(() => {
                    statusMessage.innerHTML = '';
                }, 5000);
            }
        });

        // Smooth scrolling for any internal links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });

            card.addEventListener('mouseenter', function() {
                this.style.transform = 'translateY(-8px) scale(1.02)';
            });
            
            card.addEventListener('mouseleave', function() {
                this.style.transform = 'translateY(0) scale(1)';
            });
        });
